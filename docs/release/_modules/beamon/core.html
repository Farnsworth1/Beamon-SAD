[docs]def get_rmat(M, N):
    """
    Gets the 3D Rotation matrix of a plane defined with vector normal M.
    N will be the vector normal to the plane you want to rotate into.

    :param M: 3 component list
    :type M: double
    :param N: 3 component list
    :type N: double
    :return: 3x3 matrix
    :rtype: double
    """
    # Let M be the vector normal to your current plane, and N be the vector normal to the plane you want to rotate int
    M = np.array(M)
    N = np.array(N)
    # If M == N you can stop now and leave the original points unchanged
    if np.array_equal(np.absolute(M), np.absolute(N)):
        return None
    # Calculate the rotation angle
    costheta = np.dot(M, N) / (norm(M) * norm(N))
    # Calculate the rotation axis
    axis = np.cross(M, N) / norm(np.cross(M, N))
    c = costheta
    s = np.sqrt(1 - c * c)
    C = 1 - c
    x = axis[0]
    y = axis[1]
    z = axis[2]
    return np.array([[x * x * C + c, x * y * C - z * s, x * z * C + y * s],
                     [y * x * C + z * s, y * y * C + c, y * z * C - x * s],
                     [z * x * C - y * s, z * y * C + x * s, z * z * C + c]])




[docs]def transform(T, points):
    """
    Transforms each point in points matrix using the transformation matrix T.
    points should have row wise point coordinates.

    :param T: 3x3 Transformation Matrix
    :type T: double
    :param points: nx3 points matrix
    :type points: double
    :return: nx3 points matrix
    :rtype: double
    """
    out = np.zeros_like(points, dtype=np.double)
    for i in range(len(points)):
        out[i] = T.dot(points[i])
    return out




[docs]def make_joints(edof, nodes, dof, ndofs, bc):
    """Extends edof, nodes, dof and bc matrices with new cloned nodes that has new dof.
    This will adjust translation conditions between elements.
    New nodes and dofs will be appended at the ende of each input 'nodes' and 'dof'.
    edof matrix contain element dof numbers for each nodal point end. Free dof number is predefined as zero.
    nodes matrix contain [x,y,z] nodes coordinates in each row.
    dof matrix contain nodes degrees of freedom for each node in a row.
    bc array contain nodes degrees of freedom that are locked.
    map contains mapping information from cloned nodes to origin nodes

    :param edof: topology matrix
    :type edof: integer matrix
    :param nodes: nodes coordinates matrix
    :type nodes: float matrix
    :param dof: nodes dof matrix
    :type dof: integer matrix
    :param ndofs: number of dof in each node
    :type ndofs: integer (3 or 6)
    :param bc: boundary conditions array
    :type bc: integer array
    :param map: origin and cloned nodes mapping
    :type map: dict
    :return: edof, nodes, dof, map
    """
    el_idx = 0
    last_dof = dof[-1, -1]
    map = {}  # mapping cloned nodes to origin nodes

    for element in edof:
        # something free at element beginning node
        if len(np.where(element[2:8] == 0)[0]) != 0:
            # add a new node
            o_idx = int(element[0]) - 1  # origin node index
            nodes = np.vstack((nodes, nodes[o_idx]))  # append new node coordinates
            # map cloned node index to origin node
            map[len(nodes)] = o_idx
            n_dof = np.array([last_dof + i for i in range(1, ndofs + 1)])  # get new dof
            last_dof = n_dof[-1]
            # replace 0 values in edof
            n_dof_idx = 0
            for id in np.where(element[2:8] == 0)[0]:
                edof[el_idx, id + 2] = n_dof[id]
                n_dof_idx += 1
            # adjust boundary conditions
            # get origin node dof
            o_dof = dof[int(element[0]) - 1, :]
            # append new dof numbers to bc if origin dof numbers happen to be in bc
            for i in n_dof[np.intersect1d(bc, o_dof, assume_unique=True, return_indices=True)[2]]:
                bc.append(i)
        # something free at element end node
        if len(np.where(element[8:] == 0)[0]) != 0:
            # add a new node
            o_idx = int(element[1]) - 1  # origin node index
            nodes = np.vstack((nodes, nodes[o_idx]))  # append new node coordinates
            # map cloned node index to origin node
            map[len(nodes)] = o_idx
            n_dof = np.array([last_dof + i for i in range(1, ndofs + 1)])  # get new dof
            last_dof = n_dof[-1]
            # replace 0 values in edof
            n_dof_idx = 0
            for id in np.where(element[8:] == 0)[0]:
                edof[el_idx, id + 7] = n_dof[id]
                n_dof_idx += 1
            # adjust boundary conditions
            # get origin node dof
            o_dof = dof[int(element[1]) - 1, :]
            # append new dof numbers to bc if origin dof numbers happen to be in bc
            for i in n_dof[np.intersect1d(bc, o_dof, assume_unique=True, return_indices=True)[2]]:
                bc.append(i)
        el_idx += 1

    # make new dof matrix
    dof = cal.createdofs(len(nodes), ndofs)
    return edof, nodes, dof, bc, map




[docs]def get_local_orientation(orientation):
    """
    Calculates all local axes orientation vectors according to v vector and
     normal vectors n = (node2-node1)/|node2-node1|
    Z axes is the projection of v on the plane with orthogonal vector X axes

    :return: x,y,z directions in each row
    """
    # get all n and v vectors
    # orientation = self.data_set.get_link_orientations()
    x = np.zeros((len(orientation), 3))  # x axes vectors
    y = np.zeros((len(orientation), 3))  # y axes vectors
    z = np.zeros((len(orientation), 3))  # z axes vectors
    index = 0
    for row in orientation:
        x_el = np.array([row[0], row[1], row[2]])
        v = np.array([row[3], row[4], row[5]])  # direction vector
        # normalize
        x_el = normalize(x_el.reshape((1, -1)))[0]
        v = normalize(v.reshape((1, -1)))[0]
        z_el = _project_onto_plane(v, x_el)
        y_el = -np.cross(x_el, z_el)  # negative because left hand coordinate systems are being used
        x[index] = x_el
        y[index] = y_el
        z[index] = z_el
        index += 1
    return x, y, z




[docs]def assemble_global_f(loads, Dof):
    """Assemble global loads vector from nodes loads.
    Loads should contain nodes coordinates and the forces applied to each node [x,y,z, u,v,w, m_x,m_y,m_z]

    :param loads: Nodes loads array
    :type loads: float matrix
    :param Dof: nodes degrees of freedom
    :type Dof: integer matrix
    :return: global force vector (n x 1 vector)
    :rtype: float vector
    """
    # global f size is the last dof number
    f = np.zeros([Dof[-1, -1], 1])
    for load in loads:
        n_dofs = Dof[int(load[0]) - 1, :] - 1  # get dof index in global f for each node
        column_id = 0
        for dof_id in n_dofs:
            f[dof_id] += load[column_id + 1]  # superposition of each load vector component
            column_id += 1
    return f




[docs]def assemble_element_loads(loads, Edof):
    """Assemble element local loads vector from element loads.
    Loads should contain [link_id, qx, qy, qz, qw]

    :param loads: element loads
    :type loads: float matrix
    :param Edof: Element degree of freedom (Topology matrix)
    :type Edof: integer matrix
    :return: local forces
    :rtype: float matrix
    """
    eq = np.zeros([len(Edof), 4])
    for load in loads:
        eq[int(load[0]) - 1] = load[1:]
    return eq




[docs]def assemble_bc(nodes_bc, Dof):
    """Assemble boundary condition information in a vector.
    nodes_bc should contain [u_x,u_y,u_z,phi_x,phi_y,phi_z] for each node, which can only contain
    0 (free) or 1 (locked).
    The assembled bc vector could be ex. [1,2,5,6] that locks movement in those numbered directions.

    :param nodes_bc: boundary condition information from database
    :type nodes_bc: integer matrix
    :param Dof: nodes degrees of freedom
    :type Dof: integer matrix
    :return: bc vector
    :rtype: integer 1 x n vector
    """
    bc = []
    row_id = 0
    for bcond in nodes_bc:
        indices = np.where(bcond == 1)[0]
        for i in indices:
            bc.append(Dof[row_id, i])
        row_id += 1
    return np.array(bc)




[docs]def assemble_nodes_results(o_edof, nodes, a, r, map):
    """Assemble nodes displacements and support forces according to joints mapping into a pandas dataframe.

    :param o_edof: origin edof
    :type o_edof: integer matrix
    :param nodes: nodes coordinates
    :type nodes: float matrix
    :param a: nodes displacements
    :type a: float vector
    :param r: nodes support forces
    :type r: float vector
    :param map: origin nodes to cloned joint nodes map
    :type map: dict
    """
    a = np.reshape(np.array(a), (-1, 6))  # displacements
    Q = np.array(r).reshape((-1, 6))  # reaction forces
    results = np.zeros((len(Q), 16))
    idx = 0
    node_idx = 0
    for n_a, n_q, n_coord in zip(a, Q, nodes):
        if idx > len(o_edof):
            node_idx = map[idx]  # if joint node take origin node index
        else:
            node_idx += 1  # if origin node just take it's index
        results[idx] = np.hstack((node_idx, n_a, n_coord, n_q))
        idx += 1
    # values will be sorted for Node Nr. in case of existing joint nodes that has the same number as their origin nodes
    return pd.DataFrame(results,
                        columns=['Node Nr.', 'ax', 'ay', 'az', 'aphi_x', 'aphi_y', 'aphi_z', 'x', 'y', 'z', 'Qx', 'Qy',
                                 'Qz', 'Qphi_x', 'Qphi_y', 'Qphi_z']).sort_values('Node Nr.')



def _project_onto_plane(v, n):
    """gets the projection of v onto a plane defined by n as its orthogonal vector.

    :param v: numpy vector in R3
    :param n: numpy normal vector in R3
    :return: numpy array
    """
    d = np.dot(v, n) / np.linalg.norm(n)
    p = [d * n[i] for i in range(len(n))]
    return [v[i] - p[i] for i in range(len(v))]
