    def nextClicked(self):
        if self.valid_input:
            # upload inputs to table
            self.ui.tableWidget.item(0, 0).setText(self.ui.lineEdit_name.text())
            self.ui.tableWidget.item(0, 1).setText(self.parent.ui.lineEditys.text())
            self.ui.tableWidget.item(0, 2).setText(self.parent.ui.lineEditzs.text())
            self.ui.tableWidget.item(0, 3).setText(self.parent.ui.lineEdityM2.text())
            self.ui.tableWidget.item(0, 4).setText(self.parent.ui.lineEditzM2.text())
            self.ui.tableWidget.item(0, 5).setText(self.parent.ui.lineEditA.text())
            self.ui.tableWidget.item(0, 6).setText(self.parent.ui.lineEditIy.text())
            self.ui.tableWidget.item(0, 7).setText(self.parent.ui.lineEditIz.text())
            self.ui.tableWidget.item(0, 8).setText(self.parent.ui.lineEditIyz.text())
            self.ui.tableWidget.item(0, 9).setText(self.parent.ui.lineEditIT.text())
            self.ui.tableWidget.item(0, 10).setText(self.ui.lineEdit_kz.text())
            self.ui.tableWidget.item(0, 11).setText(
                str(float(self.ui.lineEdit_kz.text()) * float(self.parent.ui.lineEditA.text())))

        else:
            self.ui.wizardPage1.setSubTitle(self.warning_text)
            self.restart()

    def finishedClicked(self):
        section = {'name': self.ui.lineEdit_name.text(),
                   'y_sp': float(self.parent.ui.lineEditys.text()),
                   'z_sp': float(self.parent.ui.lineEditzs.text()),
                   'y_smp': float(self.parent.ui.lineEdityM2.text()),
                   'z_smp': float(self.parent.ui.lineEditzM2.text()),
                   'A': float(self.parent.ui.lineEditA.text()),
                   'Iy': float(self.parent.ui.lineEditIy.text()),
                   'Iz': float(self.parent.ui.lineEditIz.text()),
                   'Iyz': float(self.parent.ui.lineEditIyz.text()),
                   'J': float(self.parent.ui.lineEditIT.text()),
                   'kz': float(self.ui.lineEdit_kz.text()),
                   'AQz': float(self.ui.lineEdit_kz.text()) * float(self.parent.ui.lineEditA.text())}  # AQz = A * kz
        # save in the database
        self.parent.parent.database.add_section(section)
        # refresh section table if open
        if not self.parent.parent.database.visualization is None:
            if self.parent.parent.visualizer.state == 'section':
                section_table = self.parent.parent.ui.tableDockWidget.widget()
                section_table.clearTable()
                section_table.load_table()


    def validateName(self, input):
        if ' ' in input:
            self.valid_input = False
            self.warning_text = "Name der Sektion sollte keine Leerzeichen enthalten"
            self.ui.label_warning.setText(self.warning_text)
            self.next_button.setEnabled(False)
        elif self.parent.parent.database.contains_section(name=input):
            self.valid_input = False
            self.warning_text = "Name der Sektion ist schon Verhanden"
            self.ui.label_warning.setText(self.warning_text)
            self.next_button.setEnabled(False)
        else:
            self.valid_input = True
            self.next_button.setEnabled(True)
            self.ui.label_warning.setText("")
