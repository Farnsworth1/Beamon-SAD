[docs]    def initPlotWidget(self):
        """
        initializes the plot area settings and add the ROI elements to the plotwidget
        """

        self.ui.plotWidget.setMenuEnabled(False)

        # very important in order to prevent sheering
        self.plotter.plotItem.getViewBox().setAspectLocked()

        # text = """Überlappungsmodell (Dünnwandig)"""
        # self.ui.plotWidget.getPlotItem().setTitle(text)

        # dont't show axes since they visible in grid
        self.plotter.hideAxis('bottom')
        self.plotter.hideAxis('left')

        # make the drawing color of the elements black
        self.pen = pg.mkPen('k')
        # add a grid
        self.grid = pg.GridItem(self.pen)
        self.plotter.addItem(self.grid)
        # add plot items related to each cross section case
        self.roi1 = None
        self.roi2 = None
        self.text1 = None
        self.text2 = None
        self.text3 = None

        # TODO: add y,z axis
        self.update_global()


    def update_global(self):
        tab_name = self.ui.tabWidget.currentWidget().objectName()
        self.remove_all_items()
        if tab_name == "tab1":
            # rectangular
            # get parameters
            try:
                b = float(self.ui.In11.text())
                d = float(self.ui.In12.text())
            except Exception:
                b = 1
                d = 1
            # create objects
            self.roi1 = pg.RectROI(pos=[0, 0], size=[b, d], pen=self.pen, scaleSnap=True,
                                   snapSize=0.1, translateSnap=True)
            self.roi1.handlePen = self.pen
            self.roi1.addScaleHandle([0.5, 1], center=[0.5, 0])
            self.roi1.addScaleHandle([1, 0.5], center=[0, 0.5])
            self.text1 = pg.TextItem("b", color='k')
            self.text2 = pg.TextItem("d", color='k')
            self.text1.setPos(self.roi1.pos()[0] + b / 2, self.roi1.pos()[1])
            self.text2.setPos(self.roi1.pos()[0], self.roi1.pos()[1] + d / 2)
            self.roi1.sigRegionChanged.connect(self.roi1_region_changed)
            # add objects to plotter
            self.plotter.addItem(self.roi1)
            self.plotter.addItem(self.text1)
            self.plotter.addItem(self.text2)
        if tab_name == "tab2":
            # triangle
            try:
                a = float(self.ui.In21.text())
                b = float(self.ui.In22.text())
            except:
                a = 1
                b = 1
            if a*2 <= b:
                a = 0
                b = 0
            h = np.sqrt((a ** 2) - (b * b / 4))
            # create objects
            self.roi2 = pg.PolyLineROI([[0, 0], [b / 2, h], [b, 0]], closed=True, pen=self.pen, movable=False)
            self.text1 = pg.TextItem("a", color='k')
            self.text2 = pg.TextItem("a", color='k')
            self.text3 = pg.TextItem("b", color='k')
            self.text1.setPos(b / 4, h / 2)
            self.text2.setPos(3 * b / 4, h / 2)
            self.text3.setPos(b / 2, 0)
            self.plotter.addItem(self.roi2)
            self.plotter.addItem(self.text1)
            self.plotter.addItem(self.text2)
            self.plotter.addItem(self.text3)
        self.plotter.enableAutoRange()
        # calculate results parameters
        self.update_results(tab_name)

    def update_results(self, tab_name):
        A = 0
        Iy = 0
        Iz = 0
        Kv = 0
        if tab_name == "tab1":
            # rectangular
            try:
                b = float(self.ui.In11.text())
                d = float(self.ui.In12.text())
            except Exception:
                return

            A = b * d
            Iy = (b * d * d * d) / 12
            Iz = (d * b * b * b) / 12
            if b <= d and b != 0:
                Kv = b * (d ** 3) * ((1 / 3) - (0.21 * d / b) * (1 - (d ** 4) / (12 * b ** 4)))
            else:
                Kv = 0
            self.ui.A1.setText(str(A))
            self.ui.Iy1.setText(str(Iy))
            self.ui.Iz1.setText(str(Iz))
            self.ui.Kv1.setText(str(Kv))
        if tab_name == "tab2":
            # triangle
            try:
                a = float(self.ui.In21.text())
                b = float(self.ui.In22.text())
            except:
                return
            if a*2 <= b:
                a=0
                b=0
            h = np.sqrt((a ** 2) - (b * b / 4))
            A = b * h / 2
            Iy = b * (h ** 3) / 36
            Iz = h * (b ** 3) / 48
            if a == b:
                Kv = (h ** 4) / 26
            self.ui.h2.setText(str(h))
            self.ui.A2.setText(str(A))
            self.ui.Iy2.setText(str(Iy))
            self.ui.Iz2.setText(str(Iz))
            self.ui.Kv2.setText(str(Kv))

    def roi1_region_changed(self):
        b = str(self.roi1.size()[0])
        d = str(self.roi1.size()[1])
        self.ui.In11.setText(b)
        self.ui.In12.setText(d)
        self.update_results(self.ui.tabWidget.currentWidget().objectName())
        self.text1.setPos(self.roi1.pos()[0] + self.roi1.size()[0] / 2, self.roi1.pos()[1])
        self.text2.setPos(self.roi1.pos()[0], self.roi1.pos()[1] + self.roi1.size()[1] / 2)

    def remove_all_items(self):
        self.plotter.removeItem(self.roi1)
        self.plotter.removeItem(self.roi2)
        self.plotter.removeItem(self.text1)
        self.plotter.removeItem(self.text2)
        self.plotter.removeItem(self.text3)

    def save_clicked(self):
        tab_name = self.ui.tabWidget.currentWidget().objectName()
        A = 0
        Iy = 0
        Iz = 0
        Kv = 0
        if tab_name == "tab1":
            # save rectangular
            A = float(self.ui.A1.text())
            Iy = float(self.ui.Iy1.text())
            Iz = float(self.ui.Iz1.text())
            Kv = float(self.ui.Kv1.text())
        if tab_name == "tab2":
            # save triangle
            A = float(self.ui.A2.text())
            Iy = float(self.ui.Iy2.text())
            Iz = float(self.ui.Iz2.text())
            Kv = float(self.ui.Kv2.text())
        self.saveResults(A, Iy, Iz, Kv)
