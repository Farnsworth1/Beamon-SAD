[docs]class MplWidget(FigureCanvasQTAgg):

    def __init__(self, parent, width=5, height=4, dpi=100):
        self.parent = parent
        self.fig = Figure(figsize=(width, height), dpi=dpi)

        self.axes = self.fig.add_subplot(111, xlabel='x', ylabel='z', picker=True)  # projection='3d'
        super(MplWidget, self).__init__(self.fig)

        self.axes.figure.canvas.mpl_connect('pick_event', self.picker_callback)
        self.fig.tight_layout()

        self.init_settings()
        self.init_plot()


[docs]    def init_settings(self):
        """
        init default settings and states
        """
        ## Nodes variables

        # stats
        self.node_text_state = False  # status: show/hide index
        self.nodes_radius = 0.5  # the radius of the nodes distance*0.5/1
        self.nodes_radius_type = "automatic"  # status: calculation mode

        # data
        self.nodes_text = []  # contains all nodes indexes as 3D Text objects
        self.highlighted_node = None  # for highlighting nodes and getting information

        ## links variable

        # stats
        self.link_text_state = False  # how/hide index
        self.connections_removed = False
        # for drawing local coord. systems
        self.link_system_state_begin = False  # show/hide local system
        self.link_system_state_middle = False  # show/hide local system
        self.link_system_state_end = False  # show/hide local system

        # data
        self.links_text = []  # contains all links indexes as 3D Text objects
        # for drawing links
        self.first_node = None  # first picked node index
        self.second_node = None  # second picked node index
        self.connections = None
        self.link_surface = None

        ## grid variables

        # stats
        self.grid_type = "automatic"  # the calculation mode of the grid variables

        # data
        self.grid_opacity = 0.3  # opacity of the grid
        # size of the square grid
        self.grid_size = 20  # the minimum of the gird size
        self.grid_tick = 1  # the tick length of the grid
        self.minimum_gird_tick = 0.1

        ## supports variables
        self.supports = {}  # an empty dict of supports and nodes indexes

        ## node forces variables

        # stats
        self.node_forces_state = False
        self.node_forces_text_state = False  # show/hide values

        # forces variables
        self.node_forces = None  # all quiver3d plot elements
        self.node_momentum = {}  # an empty dict of momentum arrows
        self.node_forces_text = []  # all text3d plot elements
        self.node_momentum_text = []  # all text3d plot elements for momentum



[docs]    def init_plot(self):
        """
        Initiates the plots vital objects like the grid and the coordinate system
        """

        self.set_dimension(3)

        # init a 2d Grid for drawing
        size = self.grid_size / 2

        xdata = np.arange(0, size, self.grid_tick)
        xx, zz = np.meshgrid(xdata, xdata)

        self.axes.grid(alpha=self.grid_opacity)

        # self.axes.scatter(xx, zz)

    def picker_callback(self, event):
        if isinstance(event.artist, Axes3D):
            print(event.artist.eye)
        else:
            x = int(round(event.artist.xdata))
            y = int(round(event.artist.ydata))
            print("x = " + str(x))
            print("y = " + str(y))


    def set_dataset(self, dataset):
        self.dataset = dataset

    def set_dimension(self, dimension):
        self.dim = dimension
        if self.dim == 2:
            # fix aspect ration (equal for all exes and axes limits will be adjustable)
            self.axes.set_aspect(aspect='equal', adjustable='datalim')
        else:
            self.fig.clear()
            self.axes = self.fig.add_subplot(111,in_layout=True, xlabel='x', ylabel='y', zlabel='z', picker=True, projection='3d')
            self.axes.set_aspect(aspect='auto', adjustable='datalim')
